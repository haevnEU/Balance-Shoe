220 void updateBatteryState(){

222  akkuValue = analogRead(AKKU_PIN);
223 
224  // float volt = (akkuValue / (4096 / 4.2));// * 2; // gemessenen Wert am Pin umrechnen in volt
225  int volt_i = map(akkuValue, 0, 4096, 0, 210);
226  float volt = ((float)volt_i / 100) * 2;
227 
228  int ladezustand = (volt / 4.2) * 100; // volt in % umrechnen
229  device.lade_status = ladezustand;
230 
231  if(volt > 3.5){ // Bis zu 3.5V bleibt die LED gruen
232  digitalWrite(LED_GRUEN, HIGH);
233  digitalWrite(LED_ROT, LOW);
234  }
235  if(volt <=3.5){ // Ab einer Spannung von 3.5V geht die rote LED an
236  digitalWrite(LED_GRUEN, LOW);
237  digitalWrite(LED_ROT, HIGH);
238  }
239 
240  if(volt >= 4.2){
241  while(volt >= 4.2){
242  digitalWrite(LED_GRUEN, LOW);
243  digitalWrite(LED_ROT, HIGH);
244  delay(300);
245  digitalWrite(LED_GRUEN, HIGH);
246  digitalWrite(LED_ROT, LOW);
247  delay(300);

249  volt = (float)volt_i / 100 * 2;
250  }
251  }
252
