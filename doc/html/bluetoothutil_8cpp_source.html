24 char haevn::esp::bluetooth::util::calcCheckSum(char* data){
25 
26  /* Code from https://en.wikipedia.org/wiki/Fletcher%27s_checksum#Fletcher-16 */
27  uint16_t sum1 = 0;
28  uint16_t sum2 = 0;
29  int index;
30  for ( index = 0; index < 5; ++index ) {
31  sum1 = (sum1 + data[index]) % 255;
32  sum2 = (sum2 + sum1) % 255;
33  }
34  return static_cast<char>((sum2 << 8) | sum1);
35 }
